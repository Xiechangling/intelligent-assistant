  <action>
创建新的 Composer 组件（如果不存在）或增强现有组件：

```typescript
import { Send, Paperclip } from 'lucide-react'
import { useAppShellStore } from '../state/appShellStore'
import { AttachmentList } from './AttachmentList/AttachmentList'
import { VoiceInput } from './VoiceInput/VoiceInput'

export function Composer() {
  const {
    draftPrompt,
    setDraftPrompt,
    attachments,
    addFileAttachments,
    submitPrompt,
    assistantStatus,
  } = useAppShellStore()

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    if (draftPrompt.trim() && assistantStatus === 'idle') {
      submitPrompt().catch(() => undefined)
    }
  }

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault()
      handleSubmit(e)
    }
  }

  const showSendButton = draftPrompt.trim().length > 0

  return (
    <div className="composer">
      {attachments.length > 0 && (
        <div className="composer__attachments">
          <AttachmentList />
        </div>
      )}
      
      <form onSubmit={handleSubmit} className="composer__form">
        <div className="composer__input-wrapper">
          <button
            type="button"
            onClick={addFileAttachments}
            className="composer__icon-button"
            aria-label="Attach files"
          >
            <Paperclip size={20} />
          </button>

          <textarea
            value={draftPrompt}
            onChange={(e) => setDraftPrompt(e.target.value)}
            onKeyDown={handleKeyDown}
            placeholder="Message Claude..."
            className="composer__textarea"
            rows={1}
            disabled={assistantStatus === 'streaming'}
          />

          <VoiceInput />

          {showSendButton && (
            <button
              type="submit"
              className="composer__send-button"
              aria-label="Send message"
              disabled={assistantStatus === 'streaming'}
            >
              <Send size={20} />
            </button>
          )}
        </div>
      </form>
    </div>
  )
}
```
