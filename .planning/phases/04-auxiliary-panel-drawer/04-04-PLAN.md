```typescript
// In RightPanel.tsx
import { Search, X } from 'lucide-react';

const [searchQuery, setSearchQuery] = useState('');

const matchesSearch = (text: string) => {
  if (!searchQuery) return true;
  return text.toLowerCase().includes(searchQuery.toLowerCase());
};

const showConnectionSettings = matchesSearch('connection api key base url endpoint');
const showPresets = matchesSearch('presets workspace conversation model');
const showKeyboardShortcuts = matchesSearch('keyboard shortcuts keybindings ctrl');
const showWorkflowCapabilities = matchesSearch('workflow capabilities skills');

// In JSX (settings view):
<div className="right-panel__search-container">
  <Search size={16} className="right-panel__search-icon" />
  <input
    type="text"
    className="right-panel__search"
    placeholder="Search settings..."
    value={searchQuery}
    onChange={(e) => setSearchQuery(e.target.value)}
  />
  {searchQuery && (
    <button 
      className="right-panel__search-clear"
      onClick={() => setSearchQuery('')}
    >
      <X size={14} />
    </button>
  )}
</div>

{showConnectionSettings && (
  <div className="right-panel__section">
    {/* Connection settings */}
  </div>
)}
```
