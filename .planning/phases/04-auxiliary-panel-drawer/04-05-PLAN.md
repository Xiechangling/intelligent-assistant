import { test, expect } from '@playwright/test';

test.describe('Drawer panel', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:1420');
    await page.waitForSelector('.app-shell');
  });

  test('right panel is closed by default', async ({ page }) => {
    const appShell = page.locator('.app-shell');
    await expect(appShell).not.toHaveClass(/app-shell--drawer-open/);
  });

  test('right panel opens when settings button clicked', async ({ page }) => {
    // Find and click settings button
    const settingsButton = page.locator('.toolbar__action[title*="Settings"]').or(
      page.locator('button:has-text("Settings")')
    );
    await settingsButton.click();

    // Verify drawer is open
    const appShell = page.locator('.app-shell');
    await expect(appShell).toHaveClass(/app-shell--drawer-open/);
    
    const rightPanel = page.locator('.right-panel');
    await expect(rightPanel).toBeVisible();
  });

  test('right panel closes when Esc pressed', async ({ page }) => {
    // Open settings
    const settingsButton = page.locator('.toolbar__action[title*="Settings"]').or(
      page.locator('button:has-text("Settings")')
    );
    await settingsButton.click();
    
    // Press Escape
    await page.keyboard.press('Escape');
    
    // Verify drawer is closed
    const appShell = page.locator('.app-shell');
    await expect(appShell).not.toHaveClass(/app-shell--drawer-open/);
  });

  test('right panel closes when close button clicked', async ({ page }) => {
    // Open settings
    const settingsButton = page.locator('.toolbar__action[title*="Settings"]').or(
      page.locator('button:has-text("Settings")')
    );
    await settingsButton.click();
    
    // Click close button
    const closeButton = page.locator('.right-panel__close');
    await closeButton.click();
    
    // Verify drawer is closed
    const appShell = page.locator('.app-shell');
    await expect(appShell).not.toHaveClass(/app-shell--drawer-open/);
  });

  test('right panel resize handle is visible', async ({ page }) => {
    // Open settings
    const settingsButton = page.locator('.toolbar__action[title*="Settings"]').or(
      page.locator('button:has-text("Settings")')
    );
    await settingsButton.click();
    
    // Verify resize handle exists
    const resizeHandle = page.locator('.right-panel__resize-handle');
    await expect(resizeHandle).toBeVisible();
  });

  test('right panel width persists across reloads', async ({ page }) => {
    // Open settings
    const settingsButton = page.locator('.toolbar__action[title*="Settings"]').or(
      page.locator('button:has-text("Settings")')
    );
    await settingsButton.click();
    
    // Get stored width
    const storedWidth = await page.evaluate(() => {
      return localStorage.getItem('rightPanelWidth') || '400';
    });
    
    // Reload page
    await page.reload();
    await page.waitForSelector('.app-shell');
    
    // Open settings again
    await settingsButton.click();
    
    // Verify width matches
    const currentWidth = await page.evaluate(() => {
      return localStorage.getItem('rightPanelWidth') || '400';
    });
    
    expect(currentWidth).toBe(storedWidth);
  });
});

test.describe('Settings panel', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:1420');
    await page.waitForSelector('.app-shell');
    
    // Open settings panel
    const settingsButton = page.locator('.toolbar__action[title*="Settings"]').or(
      page.locator('button:has-text("Settings")')
    );
    await settingsButton.click();
    
    // Switch to settings tab if needed
    const settingsTab = page.locator('.right-panel__tab:has-text("Settings")');
    if (await settingsTab.isVisible()) {
      await settingsTab.click();
    }
  });

  test('settings search filters sections', async ({ page }) => {
    // Type in search
    const searchInput = page.locator('.right-panel__search');
    await searchInput.fill('keyboard');
    
    // Verify keyboard section is visible
    const keyboardSection = page.locator('.right-panel__section:has-text("Global Shortcuts")');
    await expect(keyboardSection).toBeVisible();
    
    // Verify other sections are hidden (connection section should not be visible)
    const connectionSection = page.locator('.right-panel__section:has-text("API Settings")');
    await expect(connectionSection).not.toBeVisible();
  });

  test('settings search is case-insensitive', async ({ page }) => {
    const searchInput = page.locator('.right-panel__search');
    await searchInput.fill('KEYBOARD');
    
    const keyboardSection = page.locator('.right-panel__section:has-text("Global Shortcuts")');
    await expect(keyboardSection).toBeVisible();
  });

  test('settings search clear button works', async ({ page }) => {
    const searchInput = page.locator('.right-panel__search');
    await searchInput.fill('keyboard');
    
    // Click clear button
    const clearButton = page.locator('.right-panel__search-clear');
    await clearButton.click();
    
    // Verify search is empty
    await expect(searchInput).toHaveValue('');
    
    // Verify all sections are visible
    const connectionSection = page.locator('.right-panel__section:has-text("API Settings")');
    await expect(connectionSection).toBeVisible();
    
    const keyboardSection = page.locator('.right-panel__section:has-text("Global Shortcuts")');
    await expect(keyboardSection).toBeVisible();
  });

  test('settings groups are organized', async ({ page }) => {
    // Verify group headers exist
    const connectionGroup = page.locator('.right-panel__group-header:has-text("Connection")');
    await expect(connectionGroup).toBeVisible();
    
    const workspaceGroup = page.locator('.right-panel__group-header:has-text("Workspace")');
    await expect(workspaceGroup).toBeVisible();
    
    const keyboardGroup = page.locator('.right-panel__group-header:has-text("Keyboard")');
    await expect(keyboardGroup).toBeVisible();
    
    const capabilitiesGroup = page.locator('.right-panel__group-header:has-text("Capabilities")');
    await expect(capabilitiesGroup).toBeVisible();
  });

  test('keyboard shortcut hints are visible', async ({ page }) => {
    // Verify keyboard shortcut hints exist
    const hints = page.locator('.keyboard-hint');
    const count = await hints.count();
    
    // Should have at least 3 hints (ctrl+t, ctrl+o, ctrl+e)
    expect(count).toBeGreaterThanOrEqual(3);
    
    // Verify specific hints
    const ctrlTHint = page.locator('.keyboard-hint:has-text("t")');
    await expect(ctrlTHint.first()).toBeVisible();
  });
});
